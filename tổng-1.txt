/*  ESP32 Unified Sketch - FIXED
    - DHT11 (pin 5) -> temperature & humidity
    - DS18B20 (pin 4) -> temperature C & F
    - PIR HC-SR501 (pin 2) -> motion
    - MAX30100 (I2C SDA=21, SCL=22) -> BPM + SpO2, non-blocking
    - LCD I2C 0x27 (16x2)
    - Buttons: NEXT = GPIO16, PREV = GPIO17 (use INPUT_PULLUP, active LOW)
    - Firebase: send ALL sensors every 5 seconds
*/

// -------------------- Buttons (NEXT / PREV) --------------------
const int btnNextPin = 16; 
const int btnPrevPin = 17;
unsigned long lastDebounceTimeNext = 0;
unsigned long lastDebounceTimePrev = 0;
const unsigned long debounceDelay = 50; 
int lastBtnNextState = HIGH;
int lastBtnPrevState = HIGH;

int btnPrevState = HIGH;

// Selected sensor index
enum SensorIndex { S_DHT11 = 0, S_DS18B20 = 1, S_PIR = 2, S_MAX30100 = 3 };

// Timing for Firebase send

float last_dht_hum = NAN;

float last_ds18_tempF = NAN;

// ----------------------------------------------------------------
// Buzzer wrapper using tone() for ESP32
void playBuzzerTone(int freq) {
  tone(buzzerPin, freq);
}
void stopBuzzer() {
  noTone(buzzerPin);
}

// -------------------- Button handling --------------------
void checkButtons() {
  int readingNext = digitalRead(btnNextPin);
  int readingPrev = digitalRead(btnPrevPin);

  // NEXT
  if (readingNext != lastBtnNextState) lastDebounceTimeNext = millis();
  if ((millis() - lastDebounceTimeNext) > debounceDelay) {
    if (readingNext != btnNextState) {
      btnNextState = readingNext;
      if (btnNextState == LOW) {
        currentSensor++;
        if (currentSensor > S_MAX30100) currentSensor = S_DHT11;
        lcd.clear();
        switch (currentSensor) {
          case S_DHT11: lcd.print("DHT11"); break;
          case S_DS18B20: lcd.print("DS18B20"); break;
          case S_PIR: lcd.print("PIR HC-SR501"); break;
          case S_MAX30100: lcd.print("MAX30100"); break;
        }
      }
    }
  }
  lastBtnNextState = readingNext;

  // PREV
  if (readingPrev != lastBtnPrevState) lastDebounceTimePrev = millis();
  if ((millis() - lastDebounceTimePrev) > debounceDelay) {
    if (readingPrev != btnPrevState) {
      btnPrevState = readingPrev;
      if (btnPrevState == LOW) {
        currentSensor--;
        if (currentSensor < S_DHT11) currentSensor = S_MAX30100;
        lcd.clear();
        switch (currentSensor) {
          case S_DHT11: lcd.print("DHT11"); break;
          case S_DS18B20: lcd.print("DS18B20"); break;
          case S_PIR: lcd.print("PIR HC-SR501"); break;
          case S_MAX30100: lcd.print("MAX30100"); break;
        }
      }
    }
  }
  lastBtnPrevState = readingPrev;
}

// -------------------- Read sensors --------------------
void readDHT11() {
  float t = dht_sensor.readTemperature();
  float h = dht_sensor.readHumidity();
  if (!isnan(t) && !isnan(h)) {
    last_dht_temp = t;
    last_dht_hum = h;
    Serial.printf("DHT11 -> T: %.2f C, H: %.2f %%\n", t, h);
  }
}

void readDS18B20() {
  sensors_DS18.requestTemperatures();
  float tempC = sensors_DS18.getTempCByIndex(0);
  if (tempC != DEVICE_DISCONNECTED_C) {
    last_ds18_tempC = tempC;
    last_ds18_tempF = sensors_DS18.toFahrenheit(tempC);
    Serial.printf("DS18B20 -> %.2f C | %.2f F\n", last_ds18_tempC, last_ds18_tempF);
  }
}

void readPIR() {
  int val = digitalRead(pirPin);
  last_pir_motion = (val == HIGH) ? 1 : 0;
  Serial.println(last_pir_motion ? "PIR: Motion detected" : "PIR: No motion");
}

void readMAX30100() {
  pox.update();
  float bpm = pox.getHeartRate();
  float spo2 = pox.getSpO2();
  if (!isnan(bpm)) last_bpm = bpm / 2.2;
  if (!isnan(spo2)) last_spo2 = spo2;
  Serial.printf("MAX30100 -> BPM: %.2f | SpO2: %.2f\n", last_bpm, last_spo2);

  // Buzzer non-blocking
  if (!isnan(last_bpm) && last_bpm > 120) {
    if (millis() - lastBuzzerUpdate > 5) {
      playBuzzerTone(buzzerFreq);
      if (buzzerIncreasing) {
        buzzerFreq += 10;
        if (buzzerFreq >= 1200) buzzerIncreasing = false;
      } else {
        buzzerFreq -= 10;
        if (buzzerFreq <= 600) buzzerIncreasing = true;
      }
      lastBuzzerUpdate = millis();
    }
  } else {
    stopBuzzer();
    buzzerFreq = 600;
    buzzerIncreasing = true;
  }
}

// -------------------- Firebase send --------------------
void sendAllToFirebase() {
  // DHT11
  if (!isnan(last_dht_temp))
    Firebase.RTDB.setFloat(&fbdo, "DHT11/Nhiet_do_moi_truong", last_dht_temp);
  if (!isnan(last_dht_hum))
    Firebase.RTDB.setFloat(&fbdo, "DHT11/Do_am_moi_truong", last_dht_hum);
  // DS18B20
  if (!isnan(last_ds18_tempC))
    Firebase.RTDB.setFloat(&fbdo, "DS18B20/Temp_C", last_ds18_tempC);
  if (!isnan(last_ds18_tempF))
    Firebase.RTDB.setFloat(&fbdo, "DS18B20/Temp_F", last_ds18_tempF);
  // PIR
  Firebase.RTDB.setInt(&fbdo, "PIR/Motion", last_pir_motion);
  // MAX30100
  if (!isnan(last_bpm))
    Firebase.RTDB.setFloat(&fbdo, "MAX30100/BPM", last_bpm);
  if (!isnan(last_spo2))
    Firebase.RTDB.setFloat(&fbdo, "MAX30100/SpO2", last_spo2);
}

// -------------------- LCD --------------------
void displayCurrentSensor() {
  lcd.clear();
  switch (currentSensor) {
    case S_DHT11:
      lcd.print("DHT11");
      lcd.setCursor(0,1);
      if (!isnan(last_dht_temp) && !isnan(last_dht_hum))
        lcd.printf("T:%.1fC H:%.1f%%", last_dht_temp, last_dht_hum);
      else lcd.print("No data");
      break;
    case S_DS18B20:
      lcd.print("DS18B20");
      lcd.setCursor(0,1);
      if (!isnan(last_ds18_tempC))
        lcd.printf("%.2fC %.2fF", last_ds18_tempC, last_ds18_tempF);
      else lcd.print("No data");
      break;
    case S_PIR:
      lcd.print("PIR HC-SR501");
      lcd.setCursor(0,1);
      lcd.print(last_pir_motion ? "Motion: YES" : "Motion: NO ");
      break;
    case S_MAX30100:
      lcd.print("MAX30100");
      lcd.setCursor(0,1);
      if (!isnan(last_bpm) && !isnan(last_spo2))
        lcd.printf("BPM:%.0f S:%.0f%%", last_bpm, last_spo2);
      else lcd.print("No data");
      break;
  }
}

// -------------------- Main loop --------------------
void loop() {
  checkButtons();

  // Read sensors with timing control
  static unsigned long lastDHTRead = 0;
  if (millis() - lastDHTRead > 2000 || lastDHTRead == 0) { readDHT11(); lastDHTRead = millis(); }
  static unsigned long lastDS18Read = 0;
  if (millis() - lastDS18Read > 1000 || lastDS18Read == 0) { readDS18B20(); lastDS18Read = millis(); }
  readPIR();
  readMAX30100();

  // LCD update
  static unsigned long lastLCDUpdate = 0;
  if (millis() - lastLCDUpdate > 500) { displayCurrentSensor(); lastLCDUpdate = millis(); }

  // Firebase send
  if (Firebase.ready() && signupOK && (millis() - sendDataPrevMillis > sendInterval || sendDataPrevMillis==0)) {
    sendDataPrevMillis = millis();
    sendAllToFirebase();
  }

  delay(10);
}
