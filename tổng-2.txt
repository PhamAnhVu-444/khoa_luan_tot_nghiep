// ====================== MAX30100 TASK (Core 0) ======================
void Task_ReadSensor(void *parameter) {
  (void) parameter;
  for (;;) {
    pox.update();

    if (millis() - lastPoxReport > POX_REPORT_MS) {
      heartRate = round(pox.getHeartRate());
      spo2 = round(pox.getSpO2());

      Serial.print("[MAX30100] HR = ");
      Serial.print(heartRate);
      Serial.print(" bpm | SpO2 = ");
      Serial.print(spo2);
      Serial.println(" %");

      // Bật / tắt buzzer nếu HR > 120
      buzzerActive = (heartRate > 120);

      lastPoxReport = millis();
    }
    vTaskDelay(20 / portTICK_PERIOD_MS);
  }
}

// ====================== Firebase + DHT11 + DS18B20 + PIR TASK (Core 1) ======================
// Đã sửa đổi để gửi nhịp tim nhanh hơn
void Task_SendFirebase(void *parameter) {
  (void) parameter;

  unsigned long lastEnvSend = 0;   // Timer cho nhiệt độ môi trường
  unsigned long lastHeartSend = 0; // Timer cho nhịp tim
  
  const unsigned long ENV_INTERVAL = 5000;  // 5 giây gửi DHT/DS18B20
  const unsigned long HEART_INTERVAL = 1000; // 1 giây gửi MAX30100/PIR

  for (;;) {
    unsigned long currentMillis = millis();

    // --- 1. XỬ LÝ PIR (Realtime để bật đèn LED ngay lập tức) ---
    pirState = digitalRead(pirPin);
    digitalWrite(ledPin, pirState);

    // --- 2. GỬI MAX30100 & PIR (Mỗi 1 giây) ---
    if (currentMillis - lastHeartSend > HEART_INTERVAL) {
      lastHeartSend = currentMillis;
      
      if (Firebase.ready() && signupOK) {
        // Sử dụng setIntAsync để không chặn chương trình
        Firebase.RTDB.setIntAsync(&fbdo, "MAX30100/HeartRate", heartRate);
        Firebase.RTDB.setIntAsync(&fbdo, "MAX30100/SpO2", spo2);
        Firebase.RTDB.setStringAsync(&fbdo, "PIR/Motion", pirState ? "Co nguoi" : "Khong co nguoi");
      }
    }

    // --- 3. ĐỌC & GỬI DHT11 + DS18B20 (Mỗi 5 giây) ---
    if (currentMillis - lastEnvSend > ENV_INTERVAL) {
      lastEnvSend = currentMillis;

      // Đọc DHT11
      float t = dht.readTemperature();
      float h = dht.readHumidity();
      if (!isnan(t)) dhtTemp = t;
      if (!isnan(h)) dhtHum = h;

      // Đọc DS18B20
      ds18b20.requestTemperatures();
      dsTempC = ds18b20.getTempCByIndex(0);
      if (dsTempC != DEVICE_DISCONNECTED_C) {
        dsTempF = dsTempC * 9.0 / 5.0 + 32.0;
      } else {
        dsTempC = 0;
        dsTempF = 0;
      }

      // Gửi Firebase
      if (Firebase.ready() && signupOK) {
        Firebase.RTDB.setFloatAsync(&fbdo, "DHT11/Temperature", dhtTemp);
        Firebase.RTDB.setFloatAsync(&fbdo, "DHT11/Humidity", dhtHum);
        Firebase.RTDB.setFloatAsync(&fbdo, "DS18B20/TemperatureC", dsTempC);
        Firebase.RTDB.setFloatAsync(&fbdo, "DS18B20/TemperatureF", dsTempF);
      }
    }

    // Delay ngắn để nhường CPU cho các tác vụ khác (quan trọng)
    vTaskDelay(50 / portTICK_PERIOD_MS);
  }
}

// ====================== Buzzer Task ======================
void buzzerTask(void * parameter) {
  (void) parameter;
  pinMode(buzzerPin, OUTPUT);
  while (true) {
    if (buzzerActive) {
      tone(buzzerPin, 1500);
    } else {
      noTone(buzzerPin);
    }
    vTaskDelay(100 / portTICK_PERIOD_MS);
  }
}

// ====================== LCD & RGB Helpers ======================
unsigned long lastLCDUpdate = 0;
const unsigned long lcdInterval = 400;

void displaySensorRealtime() {
  switch(sensorIndex){
    case 0: // MAX30100
      lcd.setCursor(0,0);
      lcd.print("MAX30100 HR/SpO2  ");
      lcd.setCursor(0,1);
      lcd.print("HR:");
      if (heartRate < 10) lcd.print(" ");
      lcd.print(heartRate);
      lcd.print(" bpm ");
      if (spo2 < 10) lcd.print(" ");
      lcd.print(spo2);
      lcd.print("%  ");
      break;

    case 1: // DHT11
      lcd.setCursor(0,0);
      lcd.print("DHT11 Temp/Hum   ");
      lcd.setCursor(0,1);
      lcd.print(dhtTemp,1);
      lcd.print("C ");
      lcd.print(dhtHum,1);
      lcd.print("%    ");
      break;

    case 2: // DS18B20
      lcd.setCursor(0,0);
      lcd.print("DS18B20 Temp     ");
      lcd.setCursor(0,1);
      lcd.print(dsTempC,1);
      lcd.print("C ");
      lcd.print(dsTempF,1);
      lcd.print("F    ");
      break;

    case 3: // PIR
      lcd.setCursor(0,0);
      lcd.print("PIR Sensor       ");
      lcd.setCursor(0,1);
      if (pirState) lcd.print("Co nguoi         ");
      else lcd.print("Khong co nguoi   ");
      break;
  }
}

// Cập nhật LED RGB theo sensorIndex
void updateLED(int index) {
  switch(index) {
    case 0: // MAX30100 - RED
      digitalWrite(ledR,HIGH); digitalWrite(ledG,LOW); digitalWrite(ledB,LOW);
      break;
    case 1: // DHT11 - GREEN
      digitalWrite(ledR,LOW); digitalWrite(ledG,HIGH); digitalWrite(ledB,LOW);
      break;
    case 2: // DS18B20 - BLUE
      digitalWrite(ledR,LOW); digitalWrite(ledG,LOW); digitalWrite(ledB,HIGH);
      break;
    case 3: // PIR - CYAN
      digitalWrite(ledR,LOW); digitalWrite(ledG,HIGH); digitalWrite(ledB,HIGH);
      break;
    default:
      digitalWrite(ledR,LOW); digitalWrite(ledG,LOW); digitalWrite(ledB,LOW);
      break;
  }
}

// ====================== LCD & Button Task ======================
void Task_LCD(void * parameter) {
  (void) parameter;
  pinMode(btnNext, INPUT_PULLUP);
  pinMode(btnPrev, INPUT_PULLUP);
  pinMode(ledR, OUTPUT);
  pinMode(ledG, OUTPUT);
  pinMode(ledB, OUTPUT);

  lcd.init();
  lcd.backlight();
  lcd.clear();

  // Hiển thị và LED ban đầu
  displaySensorRealtime();
  updateLED(sensorIndex);

  bool lastNext = HIGH;
  bool lastPrev = HIGH;

  while(true){
    bool nextState = digitalRead(btnNext);
    bool prevState = digitalRead(btnPrev);

    if(nextState==LOW && lastNext==HIGH){
      vTaskDelay(30 / portTICK_PERIOD_MS);
      if(digitalRead(btnNext)==LOW){
        sensorIndex = (sensorIndex + 1) % totalSensors;
        lcd.clear();
        updateLED(sensorIndex);
      }
    }
    lastNext = nextState;

    if(prevState==LOW && lastPrev==HIGH){
      vTaskDelay(30 / portTICK_PERIOD_MS);
      if(digitalRead(btnPrev)==LOW){
        sensorIndex = (sensorIndex + totalSensors -1) % totalSensors;
        lcd.clear();
        updateLED(sensorIndex);
      }
    }
    lastPrev = prevState;

    if(millis() - lastLCDUpdate > lcdInterval){
      lastLCDUpdate = millis();
      displaySensorRealtime();
    }

    vTaskDelay(20 / portTICK_PERIOD_MS);
  }
}
