#include <avr/wdt.h>

const int dataPin  = 11;  
const int latchPin = 12;  
const int clockPin = 4;   
const int digitPins[] = {5, 6, 7, 8}; 
const int buttonPin = 9; // Chân nút nhấn

byte ledChar[] = {0x3F, 0x06, 0x5B, 0x4F, 0x66, 0x6D, 0x7D, 0x07, 0x7F, 0x6F};

int count = 0;
byte currentDigits[4];
bool lastButtonState = HIGH; // Trạng thái trước đó của nút (chưa nhấn là HIGH)

void setup() {
  wdt_enable(WDTO_2S); // Bảo vệ mạch: Nếu treo 2s tự reset
  
  pinMode(dataPin, OUTPUT);
  pinMode(latchPin, OUTPUT);
  pinMode(clockPin, OUTPUT);
  pinMode(buttonPin, INPUT_PULLUP); // Kích hoạt điện trở kéo lên nội bộ

  for (int i = 0; i < 4; i++) {
    pinMode(digitPins[i], OUTPUT);
    digitalWrite(digitPins[i], HIGH);
  }
  updateBuffer(count); // Hiển thị số 0 ban đầu
}

void loop() {
  wdt_reset(); // Reset chó canh phòng

  // --- LOGIC NHẤN NÚT ---
  bool currentButtonState = digitalRead(buttonPin);

  // Phát hiện khoảnh khắc nút vừa được nhấn xuống (HIGH sang LOW)
  if (currentButtonState == LOW && lastButtonState == HIGH) {
    delay(10); // Đợi một chút để bỏ qua nhiễu cơ khí (debounce)
    if (digitalRead(buttonPin) == LOW) { // Xác nhận lại lần nữa
      count++;
      if (count > 9999) count = 0;
      updateBuffer(count); // Cập nhật bộ đệm số để hiển thị
    }
  }
  lastButtonState = currentButtonState; // Lưu lại trạng thái để so sánh vòng lặp sau

  // --- QUÉT LED (Phải chạy liên tục để LED không tắt) ---
  for (int i = 0; i < 4; i++) {
    byte pattern = ledChar[currentDigits[i]];
    if (i == 1) pattern |= 0x80; // Dấu chấm trang trí

    digitalWrite(latchPin, LOW);
    shiftOut(dataPin, clockPin, MSBFIRST, pattern);
    digitalWrite(latchPin, HIGH);

    digitalWrite(digitPins[i], LOW); 
    delay(2); // Thời gian sáng vừa đủ
    digitalWrite(digitPins[i], HIGH);
    delayMicroseconds(200); // Nghỉ cực ngắn để ổn định dòng điện
  }
}

// Hàm chia số thành từng chữ số để quét LED
void updateBuffer(int num) {
  currentDigits[0] = (num / 1000) % 10;
  currentDigits[1] = (num / 100) % 10;
  currentDigits[2] = (num / 10) % 10;
  currentDigits[3] = num % 10;
}
